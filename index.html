        function openTab(evt, tabName) {

            for (let i = 0; i < tabContents.length; i++) {

            }


            // Remove active classes from all header buttons

            const tabButtons = document.getElementsByClassName("tab-btn");

            for (let i = 0; i < tabButtons.length; i++) {

            }
